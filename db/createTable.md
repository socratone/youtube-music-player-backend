CREATE TABLE playlist (
  id         INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
  title      TEXT NOT NULL
);

CREATE TABLE video (
  id         INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
  videoId    TEXT NOT NULL,
  title      TEXT NOT NULL
);

CREATE TABLE playlist_video (
  id         INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
  pId        INTEGER NOT NULL,
  vId        INTEGER NOT NULL
);
